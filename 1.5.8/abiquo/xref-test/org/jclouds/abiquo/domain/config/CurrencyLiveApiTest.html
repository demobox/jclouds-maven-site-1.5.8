30  /**
31   * Live integration tests for the {@link Currency} domain class.
32   * 
33   * @author Susana Acedo
34   */
35  @Test(groups = "api", testName = "CurrencyLiveApiTest")
36  public class CurrencyLiveApiTest extends BaseAbiquoApiLiveApiTest
37  {
38      public void testCreateAndGet()
39      {
40          Currency currency =
41              Currency.builder(env.context.getApiContext()).name(PREFIX + "test-currency")
42                  .symbol("test-$").digits(2).build();
43          currency.save();
44  
45          Currency apiCurrency =
46              env.context.getPricingService().findCurrency(
47                  CurrencyPredicates.name(PREFIX + "test-currency"));
48          assertNotNull(apiCurrency);
49          assertEquals(currency.getName(), apiCurrency.getName());
50  
51          apiCurrency.delete();
52      }
53  
54      @Test(dependsOnMethods = "testCreateAndGet")
55      public void testUpdate()
56      {
57          Iterable<Currency> currencies = env.context.getPricingService().listCurrencies();
58          assertNotNull(currencies);
59  
60          Currency currency = currencies.iterator().next();
61          String name = currency.getName();
62  
63          currency.setName(PREFIX + "t-currency-upd");
64          currency.update();
65  
66          Currency apiCurrency =
67              env.context.getPricingService().findCurrency(
68                  CurrencyPredicates.name(PREFIX + "t-currency-upd"));
69  
70          assertNotNull(apiCurrency);
71          assertEquals(PREFIX + "t-currency-upd", apiCurrency.getName());
72  
73          currency.setName(name);
74          currency.update();
75      }
