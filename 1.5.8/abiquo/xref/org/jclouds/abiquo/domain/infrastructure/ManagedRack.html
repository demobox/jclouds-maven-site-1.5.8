98       * UCS. If the data provided for the connection is invalid a UcsRack will be created in Abiquo
99       * but with no Physical Machines attached to it.
100      * 
101      * @see API: <a href=

104      *      /a>

107     {
108         target =
109             context.getApi().getInfrastructureApi()
110                 .createManagedRack(datacenter.unwrap(), target);

591         public ManagedRack build()
592         {
593             UcsRackDto dto = new UcsRackDto();
594             dto.setId(id);
595             dto.setName(name);
596             dto.setShortDescription(shortDescription);
597             dto.setHaEnabled(haEnabled);
598             dto.setNrsq(nrsq);
599             dto.setVlanIdMax(vlanIdMax);
600             dto.setVlanIdMin(vlanIdMin);
601             dto.setVlanPerVdcReserved(vlanPerVdcReserved);
602             dto.setVlansIdAvoided(vlansIdAvoided);
603             dto.setPort(port);
604             dto.setIp(ip);
605             dto.setPassword(password);
606             dto.setUser(user);
607             dto.setDefaultTemplate(defaultTemplate);
608             dto.setMaxMachinesOn(maxMachinesOn);
609 
610             ManagedRack rack = new ManagedRack(context, dto);
611             rack.datacenter = datacenter;
612             return rack;
613         }

615         public static Builder fromRack(final ManagedRack in)
616         {
617             return ManagedRack.builder(in.context, in.datacenter).id(in.getId()).name(in.getName())
