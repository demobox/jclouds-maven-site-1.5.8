43  /**
44   * Adds high level functionality to {@link StoragePoolDto}. The Storage Pool Resource allows you to
45   * perform any administrative task for remote pools.
46   * 
47   * @author Ignasi Barrera
48   * @author Francesc Montserrat
49   * @see API: <a href="http://community.abiquo.com/display/ABI20/StoragePoolResource">
50   *      http://community.abiquo.com/display/ABI20/StoragePoolResource</a>
51   */
52  @EnterpriseEdition
53  public class StoragePool extends DomainWrapper<StoragePoolDto>
54  {
55      /** The default value for the used space. */
56      private static final long DEFAULT_USED_SIZE = 0;
57  
58      /** The datacenter where the storage device is. */
59      // Package protected to allow the storage device to be set automatically when discovering the
60      // pools in a device.
61      StorageDevice storageDevice;
62  
63      /**
64       * Constructor to be used only by the builder.
65       */
66      protected StoragePool(final RestContext<AbiquoApi, AbiquoAsyncApi> context, final StoragePoolDto target)
67      {
68          super(context, target);
69      }
70  
71      // Domain operations
72  
73      /**
74       * Delete the storage pool.
75       * 
76       * @see API: <a href=
77       *      "http://community.abiquo.com/display/ABI20/StoragePoolResource#StoragePoolResource-Deleteastoragepool"
78       *      > http://community.abiquo.com/display/ABI20/StoragePoolResource#StoragePoolResource-
79       *      Deleteastoragepool</a>
80       */
81      public void delete()
82      {
83          context.getApi().getInfrastructureApi().deleteStoragePool(target);
84          target = null;
85      }
86  
87      /**
88       * Create a storage pool. Create a storage pool means registering an existing storage pool
89       * obtained from {@link StorageDevice#listRemoteStoragePools} method and saving it. The Storage
90       * Pools must be associated with a Tier using {@link #setTier}.
91       * 
92       * @see API: <a href=
93       *      "http://community.abiquo.com/display/ABI20/StoragePoolResource#StoragePoolResource-Createastoragepoolwithatierlink"
94       *      > http://community.abiquo.com/display/ABI20/StoragePoolResource#StoragePoolResource-
95       *      Createastoragepoolwithatierlink</a>
96       */
97      public void save()
98      {
99          target =
100             context.getApi().getInfrastructureApi()
101                 .createStoragePool(storageDevice.unwrap(), target);
102     }
103 
104     /**
105      * Update pool information in the server with the data from this pool. Storage pool parameters
106      * cannot be updated by a user, so the parameters are only a representation of the remote pool.
107      * Although the whole storage pool entity is sent to the API in the update call, the only thing
108      * a user can change is the tier that the pool belongs to by calling {@link #setTier}.
109      * 
110      * @see API: <a href=
111      *      "http://community.abiquo.com/display/ABI20/Storage+Pool+Resource#StoragePoolResource-UpdateaStoragePool"
112      *      > http://community.abiquo.com/display/ABI20/Storage+Pool+Resource#StoragePoolResource-
113      *      UpdateaStoragePool</a>
114      */
115     public void update()
116     {
117         target = context.getApi().getInfrastructureApi().updateStoragePool(target);
118     }
119 
120     public void refresh()
121     {
122         target =
123             context.getApi().getInfrastructureApi()
124                 .refreshStoragePool(target, StoragePoolOptions.builder().sync(true).build());
125     }

127     /**
128      * Define the tier in which the pool will be added.
129      * 
130      * @see API: <a href=
131      *      "http://community.abiquo.com/display/ABI20/StoragePoolResource#StoragePoolResource-Createastoragepoolwithatierlink"
132      *      > http://community.abiquo.com/display/ABI20/StoragePoolResource#StoragePoolResource-
133      *      Createastoragepoolwithatierlink</a>
134      */
135     public void setTier(final Tier tier)
136     {
137         checkNotNull(tier, ValidationErrors.NULL_RESOURCE + Privilege.class);
138         checkNotNull(tier.getId(), ValidationErrors.MISSING_REQUIRED_FIELD + " id in " + Tier.class);
139 
140         this.updateLink(target, ParentLinkName.TIER, tier.unwrap(), "edit");
141     }

145     /**
146      * Get the device where the pool belongs.
147      * 
148      * @see API: <a href=
149      *      "http://community.abiquo.com/display/ABI20/StorageDeviceResource#StorageDeviceResource-Retrieveastoragedevice"
150      *      > http://community.abiquo.com/display/ABI20/StorageDeviceResource#StorageDeviceResource-
151      *      Retrieveastoragedevice</a>
152      */
153     public StorageDevice getStorageDevice()
154     {
155         RESTLink link =
156             checkNotNull(target.searchLink(ParentLinkName.STORAGE_DEVICE),
157                 ValidationErrors.MISSING_REQUIRED_LINK + " " + ParentLinkName.STORAGE_DEVICE);
158 
159         ExtendedUtils utils = (ExtendedUtils) context.getUtils();
160         HttpResponse response = utils.getAbiquoHttpClient().get(link);
161 
162         ParseXMLWithJAXB<StorageDeviceDto> parser =
163             new ParseXMLWithJAXB<StorageDeviceDto>(utils.getXml(),
164                 TypeLiteral.get(StorageDeviceDto.class));
165 
166         return wrap(context, StorageDevice.class, parser.apply(response));
167     }

184 
185         ParseXMLWithJAXB<TierDto> parser =
186             new ParseXMLWithJAXB<TierDto>(utils.getXml(), TypeLiteral.get(TierDto.class));
187 
188         return wrap(context, Tier.class, parser.apply(response));
189     }

282             dto.setAvailableSizeInMb(availableSizeInMb);
