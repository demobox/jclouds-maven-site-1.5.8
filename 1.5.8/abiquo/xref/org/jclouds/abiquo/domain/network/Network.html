337     public static Network< ? > wrapNetwork(
338         final RestContext<AbiquoApi, AbiquoAsyncApi> context, final VLANNetworkDto dto)
339     {
340         if (dto == null)
341         {
342             return null;
343         }
344 
345         Network< ? > network = null;
346 
347         switch (dto.getType())
348         {
349             case EXTERNAL:
350                 network = wrap(context, ExternalNetwork.class, dto);
351                 break;
352             case EXTERNAL_UNMANAGED:
353                 // TODO: How do we manage External && unmanaged networks ?
354                 throw new UnsupportedOperationException("EXTERNAL_UNMANAGED networks not supported yet");
355             case INTERNAL:
356                 network = wrap(context, PrivateNetwork.class, dto);
357                 break;
358             case PUBLIC:
359                 network = wrap(context, PublicNetwork.class, dto);
360                 break;
361             case UNMANAGED:
362                 network = wrap(context, UnmanagedNetwork.class, dto);
363                 break;
364         }
365 
366         return network;
367     }
368 
369     public static List<Network< ? >> wrapNetworks(
370         final RestContext<AbiquoApi, AbiquoAsyncApi> context, final List<VLANNetworkDto> dtos)
371     {
372         List<Network< ? >> networks = Lists.newLinkedList();
373         for (VLANNetworkDto dto : dtos)
374         {
375             networks.add(wrapNetwork(context, dto));
376         }
377         return networks;
378     }
