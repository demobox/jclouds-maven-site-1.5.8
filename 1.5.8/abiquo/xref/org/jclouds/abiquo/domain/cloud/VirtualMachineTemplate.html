111     public AsyncTask makePersistent(final VirtualDatacenter vdc, final Tier tier,
112         final String persistentTemplateName, final String persistentVolumeName)
113     {
114         // infrastructure
115         RESTLink storageLink = tier.unwrap().getEditLink();
116         if (storageLink == null)
117         {
118             // cloud
119             storageLink = tier.unwrap().searchLink("self");
120         }
121         storageLink.setRel(ParentLinkName.TIER);
122         return makePeristent(vdc, storageLink, persistentTemplateName, persistentVolumeName);
123     }

125     private AsyncTask makePeristent(final VirtualDatacenter vdc, final RESTLink storageLink,
126         final String persistentTemplateName, final String persistentVolumeName)
127     {
128         VirtualMachineTemplatePersistentDto persistentData =
129             new VirtualMachineTemplatePersistentDto();
130         persistentData.setPersistentTemplateName(persistentTemplateName);
131         persistentData.setPersistentVolumeName(persistentVolumeName);
132         RESTLink vdcLink =
133             new RESTLink(ParentLinkName.VIRTUAL_DATACENTER, vdc.unwrap().getEditLink().getHref());
134         RESTLink templateLink =
135             new RESTLink(ParentLinkName.VIRTUAL_MACHINE_TEMPLATE, target.getEditLink().getHref());
136 
137         persistentData.addLink(vdcLink);
138         persistentData.addLink(storageLink);
139         persistentData.addLink(templateLink);
140 
141         // SCG:
142         // A simple user should not have permissions to obtain a datacenter repository, but at this
143         // point we have the datacenter repository and enterprise ids in the own target uri. So we
144         // can obtain the path where do the POST
145         // Assumption that to create a new object a user needs to get the parent object cannot be
146         // applied in this case
147         String editUri = getURI().getPath();
148         Pattern p = Pattern.compile("\\d+");
149         Matcher m = p.matcher(editUri);
150         m.find();
151         Integer idEnt = new Integer(m.group());
152         m.find();
153         Integer idDcRepo = new Integer(m.group());
154 
155         AcceptedRequestDto<String> response =
156             context.getApi().getVirtualMachineTemplateApi()
157                 .createPersistentVirtualMachineTemplate(idEnt, idDcRepo, persistentData);
158 
159         return getTask(response);
160     }

183     public Volume getVolume()
184     {
185         if (this.isPersistent())
186         {
187             ExtendedUtils utils = (ExtendedUtils) context.getUtils();
188             HttpResponse rp =
189                 checkNotNull(utils.getAbiquoHttpClient().get(target.searchLink("volume")), "volume");
190 
191             ParseXMLWithJAXB<VolumeManagementDto> parser =
192                 new ParseXMLWithJAXB<VolumeManagementDto>(utils.getXml(),
193                     TypeLiteral.get(VolumeManagementDto.class));
194 
195             VolumeManagementDto dto = parser.apply(rp);
196             return new Volume(context, dto);
197         }
198         return null;
199     }

286     {
287         ConversionsDto convs =
288             context
289                 .getApi()
290                 .getVirtualMachineTemplateApi()
291                 .listConversions(

294                         .build());
295         return wrap(context, Conversion.class, convs.getCollection());

298     /**
299      * Starts a new conversion for a virtual machine template.
300      * 
301      * @see API: <a href=
302      *      "http://community.abiquo.com/display/ABI20/Conversion+Resource#ConversionResource-RequestConversion"
303      *      > http://community.abiquo.com/display/ABI20/Conversion+Resource#ConversionResource-
304      *      RequestConversion</a>
305      * @param diskFormat, desired target format for the request template
306      * @return The task reference to track its progress
307      */
308     public AsyncTask requestConversion(final DiskFormatType diskFormat)
309     {
310         ConversionDto request = new ConversionDto();
311         request.setTargetFormat(diskFormat);
312 
313         AcceptedRequestDto<String> taskRef =
314             context.getApi().getVirtualMachineTemplateApi()
315                 .requestConversion(target, diskFormat, request);
316 
317         return taskRef == null ? null : getTask(taskRef);
318     }
