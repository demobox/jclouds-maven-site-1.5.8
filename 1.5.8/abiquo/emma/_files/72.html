523    public AsyncTask setNics(final Network< ? > gatewayNetwork, final List<Ip< ? , ? >> ips,
524        final List<UnmanagedNetwork> unmanagetNetworks)
525    {
526        RESTLink configLink =
527            checkNotNull(target.searchLink(ParentLinkName.NETWORK_CONFIGURATIONS),
528                ValidationErrors.MISSING_REQUIRED_LINK + ParentLinkName.NETWORK_CONFIGURATIONS);
529 
530        // Remove the gateway configuration and the current nics
531        Iterables.removeIf(
532            target.getLinks(),
533            Predicates.or(LinkPredicates.isNic(),
534                LinkPredicates.rel(ParentLinkName.NETWORK_GATEWAY)));
535 
536        // Add the given nics in the appropriate order
537        int i = 0;
538        if (ips != null)
539        {
540            for (i = 0; i < ips.size(); i++)
541            {
542                RESTLink source = LinkUtils.getSelfLink(ips.get(i).unwrap());
543                RESTLink link = new RESTLink("nic" + i, source.getHref());
544                link.setType(ips.get(i).unwrap().getBaseMediaType());
545                target.addLink(link);
546            }
547        }
548 
549        // Add unmanaged network references, if given
550        if (unmanagetNetworks != null)
551        {
552            for (UnmanagedNetwork unmanaged : unmanagetNetworks)
553            {
554                RESTLink source =
555                    checkNotNull(unmanaged.unwrap().searchLink("ips"),
556                        ValidationErrors.MISSING_REQUIRED_LINK + "ips");
557 
558                RESTLink link = new RESTLink("nic" + i, source.getHref());
559                link.setType(UnmanagedIpDto.BASE_MEDIA_TYPE);
560                target.addLink(link);
561                i++;
562            }
563        }
564 
565        // Set the new network configuration
566        if (gatewayNetwork != null)
567        {
568            target.addLink(new RESTLink(ParentLinkName.NETWORK_GATEWAY, configLink.getHref() + "/"
569                + gatewayNetwork.getId()));

573    }
